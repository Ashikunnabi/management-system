# use below steps started with number if you are going to setup this sytem in a new server

# Generate migration scripts (If model changed)
python manage.py makemigrations

# Apply migrations to all schemas 
1. python manage.py migrate_schemas --shared

# Apply migrations to specific schemas 
python manage.py migrate_schemas --schema=schema_name

# load data to specific schema
python manage.py tenant_command loaddata --schema=schema_name import_sql/bootup.json


#  create new user and adding new domain for the user. 
2. from apps.core.rbac.models import Customer, Domain

# create your public tenant
3. tenant = Customer(schema_name='public',
                	name='Name',
                	address="Customer address",
                	status=1,
		)
tenant.save()

# Add one or more domains for the tenant
4. domain = Domain(domain = "localhost",
		tenant = tenant,
		is_primary = True,
		)
domain.save()

# load configuration to db
5. python manage.py tenant_command loaddata --schema=public import_sql/feature.json
