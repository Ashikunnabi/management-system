#  create new user and adding new domain for the user. 
from apps.core.rbac.models import Customer, Domain

# create your public tenant
tenant = Customer(schema_name='public',
                	name='Name',
                	address="Customer address",
                	status=1,
		)
tenant.save()

# Add one or more domains for the tenant
domain = Domain(domain = "localhost",
		sub_domain='localhost',
		customer=tenant,
		tenant = tenant,
		is_primary = True,
		)
domain.save()
